29def _headers(**kwargs): 

30 iap_auth_token = kwargs.get('iap_auth_token', None) 

31 if iap_auth_token is None: 

32 iap_auth_token = os.getenv('IAP_AUTH_TOKEN', None) 

33 assert iap_auth_token is not None, "IAP_AUTH_TOKEN is not defined" 

34 headers = { 

35 "Content-Type": "application/json", 

36 "Authorization": "Bearer {0}".format(iap_auth_token)

41def _response(response: requests.Response):

171 def paginated_list(self, **kwargs): 

172 params = _params(**kwargs) 

173 resp = requests.get(self.endpoint, params=params, headers=self.headers) 

174 page = _response(resp) 

175 

176 for item in page['items']: 

177 yield item 

178 

179 if 'nextPageToken' in page: 

180 kwargs.update(page_token=page['nextPageToken']) 

181 yield from self.paginated_list(**kwargs)
